Library UnderlyingGraph


Require Import Utf8.
Require Export Category Graph.

Set Implicit Arguments.

Generalizable All Variables.

Section underlying_graph.
  Context `(C : @Category objC).

  Definition UnderlyingGraph : Graph
    := {| Vertex := objC;
          Edge := Morphism C |}.

  Definition UnderlyingGraph' : Graph'
    := Eval cbv beta iota zeta delta [UnderlyingGraph GraphToGraph' Vertex Edge] in GraphToGraph' UnderlyingGraph.
End underlying_graph.
