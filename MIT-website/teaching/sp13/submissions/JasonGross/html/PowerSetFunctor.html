Library PowerSetFunctor


Require Import Utf8.
Require Import FunctionalExtensionality ProofIrrelevance.
Require Export Functor SetCategory Duals Ensembles.
Require Import Common Morphism.

Set Implicit Arguments.

Generalizable All Variables.

Ltac power_set_t :=
  repeat (apply Extensionality_Ensembles
                || apply FunctionalExtensionality.functional_extensionality_dep
                || intro
                || split
                || destruct_head_hnf @ex
                || destruct_head_hnf @and
                || (hnf in *; subst)
                || (repeat esplit; eassumption)).

Power Set Functor

Section PowerSetFunctor.
  Definition PowerSetFunctor : Functor CategoryOfTypes CategoryOfTypes.
    refine (Build_Functor CategoryOfTypes CategoryOfTypes
                          (fun X ⇒ Ensemble X)
                          (fun _ _ m ⇒ image m)
                          _
                          _);
    compute;
    abstract power_set_t.
  Defined.

  Definition PowerSetFunctorOp : Functor CategoryOfTypes (OppositeCategory CategoryOfTypes).
    refine (Build_Functor CategoryOfTypes (OppositeCategory CategoryOfTypes)
                          (fun X ⇒ Ensemble X)
                          (fun _ _ m ⇒ fun f ⇒ f ○ m)
                          _
                          _);
    compute;
    abstract power_set_t.
  Defined.
End PowerSetFunctor.
