Section hausdorff.
  Variable X : Type.
  Variable T : Topology X.

  Definition hausdorff := ∀ x y, x ≠ y →
                                      ∃ Ux Uy,
                                        (Open Ux) ∧ (Open Uy)
                                        ∧ (In _ Ux x) ∧ (In _ Uy y)
                                        ∧ Disjoint _ Ux Uy.
End hausdorff.
