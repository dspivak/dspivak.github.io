The zero group


Section zero_group.
There is a group with only the identity element
  Definition ZeroComputationalGroup : ComputationalGroup unit
    := {| group_monoid := {| id := tt; comp := λ _ _ : unit, tt |};
          group_inverse := λ _ : {| id := tt; comp := λ _ _ : unit, tt |}, tt |}.
This can also be defined by repeat esplit. Grab Existential Variables. repeat esplit.

  Global Instance ZeroIsMonoid : IsMonoid ZeroComputationalGroup.
  repeat (esplit || intros []).
  Qed.

  Global Instance ZeroIsGroup : IsGroup ZeroComputationalGroup.
  repeat (esplit || intros []).
  Qed.

  Definition ZeroGroup : Group unit := ZeroIsGroup.

Morphisms from the zero group

  Definition HomomorphismFromZeroGroup `(G : Group G0) : GroupHomomorphism ZeroGroup G.
    ∃ (fun _ ⇒ id);
    abstract (
        intros;
        destruct_head_hnf Group;
        destruct_head_hnf IsGroup;
        try rewrite left_id;
        reflexivity
      ).
  Defined.

They're unique.
  Lemma HomomorphismFromZeroGroup_unique `(G : Group G0)
  : ∀ F : GroupHomomorphism ZeroGroup G, F = HomomorphismFromZeroGroup G.
    intro F.
    apply MonoidHomomorphism_Eq.
    intros []; simpl.
    simpl_do_clear ltac:(fun H ⇒ rewrite H) (MonoidHomomorphism_IdentityLaw F).
    reflexivity.
  Qed.

Morphisms to the zero group

  Definition HomomorphismToZeroGroup `(G : Group G0) : GroupHomomorphism G ZeroGroup.
    ∃ (fun _ ⇒ tt);
    reflexivity.
  Defined.

They're unique.
  Lemma HomomorphismToZeroGroup_unique `(G : Group G0)
  : ∀ F : GroupHomomorphism G ZeroGroup, F = HomomorphismToZeroGroup G.
    intro F.
    apply MonoidHomomorphism_Eq.
    intros; simpl.
    apply unit_eq.
  Qed.
End zero_group.
