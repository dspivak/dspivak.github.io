Kleisli categories have coproducts

Section KleisliCoproducts.
  Context `(T : @Monad objC C).
  Let KT := KleisliCategory T.

  Local Open Scope morphism_scope.

  Global Instance KleisliCategory_has_is_coproducts
         (X Y : KT)
         `(C_has_coproduct : is_coproduct _ C X Y XuY)
         (η := MonadUnit T)
         (µ := MonadMultiplication T)
  : is_coproduct KT X Y XuY.
  refine (@Build_is_coproduct _ KT X Y XuY
                            (η _ ○ i₁)
                            (η _ ○ i₂)
                            (fun A ⇒ @coproduct_map _ C _ _ _ C_has_coproduct (T A))
                            _
                            _
                            _);
    abstract (
        pose proof (LeftIdentity KT);
        pose proof (RightIdentity KT);
        destruct T;
        clear T;
        repeat (reflexivity
                  || intro
                  || (progress subst_body)
                  || (progress (simpl in *; unfold Object in *; subst))
                  || (apply coproduct_inl_commutes)
                  || (apply coproduct_inr_commutes)
                  || (apply coproduct_map_unique)
                  || (progress rewrite_hyp)
                  || (rewrite <- Associativity))
      ).
  Defined.
End KleisliCoproducts.
