Problem

Let T = (ℙ, η, µ) be the power set monad. The category Kls(T) is closed under binary products, i.e. every pair of objects A B : Ob(Kls(T)) have a product in Kls(T). What is the product of A = {1,2,3} and B = {a, b}?

  Local Open Scope morphism_scope.

  Let T := PowerSetMonad.

  Let η := MonadUnit T.
  Let µ := MonadMultiplication T.

  Global Instance PowerSumIsProductPower X Y
  : is_product CategoryOfTypes (T X) (T Y) (T (X + Y))%type.
  refine (@Build_is_product _ CategoryOfTypes (T X) (T Y) (T (X + Y)%type)
                            (fun P x ⇒ P (inl x))
                            (fun P x ⇒ P (inr x))
                            (fun A f g a ⇒ fun x ⇒ match x with inl x ⇒ f a x | inr x ⇒ g a x end)
                            _
                            _
                            _);
    simpl;
    abstract (repeat (reflexivity
                        || (progress rewrite_rev_hyp)
                        || apply functional_extensionality_dep
                        || intros []
                        || intro)).
  Defined.

  Global Instance PowerSetMonadProducts X Y : is_product (KleisliCategory T) X Y (X + Y)%type.
  apply KleisliCategory_has_is_products with (C_has_product := PowerSumIsProductPower _ _);
    abstract (
        repeat match goal with
                 | _ ⇒ easy
                 | _ ⇒ progress (simpl in *; hnf in *; subst)
                 | _ ⇒ intro
                 | _ ⇒ split
                 | _ ⇒ apply Extensionality_Ensembles
                 | _ ⇒ apply functional_extensionality_dep
                 | _ ⇒ progress destruct_head_hnf @ArbitraryUnion
                 | _ ⇒ progress destruct_head_hnf @ex
                 | _ ⇒ progress destruct_head_hnf @and
                 | _ ⇒ progress destruct_head_hnf @sum
                 | _ ⇒ solve [ repeat (eassumption || reflexivity || eexists (inl _) || esplit) ]
                 | _ ⇒ solve [ repeat (eassumption || reflexivity || eexists (inr _) || esplit) ]
               end
      ).
  Defined.
End Exercise_5_3_3_6.
